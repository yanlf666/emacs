

;; System Resets,Interrupt,and Operating Modes, System Control Module(SYS)
;; SFRIE
	"SFRIE &= ~;
	/*
	JMBOUTIE: JTAG mailbox output interrupt enable.0b=disabled,b=enabled
	JMBINIE : JTAG mailbox input interrupt enable.0b=disabled,b=enabled
	NMIIE   : NMI pin interrupt enable. 0b=disabled,b=enabled
	VMAIE   : Vacant memory access interrupt enable.0b=disabled,b=enabled
	OFIE	   : Oscillator fault interrupt enable.0b=disabled,b=enabled
	WDTIE   : Watchdog timer interrupt enable.0b=disabled,b=enabled
	*/"

;; SFRIFG
	"SFRIFG &= ~;
	/*
	JMBOUTIFG: JTAG mailbox output interrupt flag.0b=No,b=interrupt pending
	JMBINIFG : JTAG mailbox input interrupt flag.0b=No,b=interrupt pending
	NMIIFG	 : NMI pin interrupt flag.0b=No,b=interrupt pending
	OFIFG	 : Oscillator fault interrupt flag.0b=No,b=interrupt pending
	WDTIFG	 : Watchdog timer interrupt flag.0b=No,b=interrupt pending
	*/"

;; SFRRPCR
	"SFRRPCR &= ~;
	/*
	SYSRSTRE : Reset pin resistor enable. 0b=disabled,b=Pullup or pulldown
		   resistor at the RST/NMI pin is enabled.
	SYSRSTUP : Reset resistor pin pullup or pulldown.0b=pulldown,b=pullup
	SYSNMIIES: NMI edge select.0b=NMI on rising edge,b=NMI on falling edge
	SYSNMI 	 : NMI select.0b=Reset function,b=NMI function
	*/"
